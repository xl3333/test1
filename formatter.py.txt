def clean_text(text):
    """清理文本：去除多余空格、空行"""
    text = text.strip()
    text = ' '.join(text.split())
    lines = [line.strip() for line in text.splitlines() if line.strip()]
    return '\n'.join(lines)

def count_words(text):
    """统计文本字数（进阶功能）"""
    words = text.split()
    return len(words)

def add_title(text, title):
    """给文本添加自定义标题（进阶功能）"""
    return f"# {title}\n\n{text}"

def read_and_format(file_path, title=None):
    """读取文件并格式化，支持添加标题和字数统计"""
    try:
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
        cleaned = clean_text(content)
        if title:
            cleaned = add_title(cleaned, title)
        word_count = count_words(cleaned)
        print(f"格式化后的内容：\n{cleaned}\n\n字数统计：{word_count}")
        return cleaned
    except FileNotFoundError:
        print(f"错误：文件 {file_path} 不存在")
        return None
