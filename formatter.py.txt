import re
from docx import Document
def clean_text(text):
    """
    清理文本：去除多余空格、多余空行、行首行尾空格
    修复：保留中文和英文单词之间的正常空格
    """
    # 1. 去掉行首行尾空格
    text = text.strip()

    # 2. 将多个空格、制表符替换为单个空格
    text = re.sub(r'[ \t]+', ' ', text)

    # 3. 按行分割，清理每一行
    lines = text.splitlines()
    cleaned_lines = []

    for line in lines:
        line = line.strip()
        if line:  # 只保留非空行
            cleaned_lines.append(line)

    # 4. 用换行符连接
    return '\n'.join(cleaned_lines)


def count_words(text):
    """
    统计文本字数（进阶功能）
    修复：支持中英文混合统计
    """
    # 匹配中文汉字和英文单词
    words = re.findall(r'[\u4e00-\u9fff]+|[a-zA-Z]+', text)
    return len(words)


def add_title(text, title, level=1):
    """
    给文本添加自定义标题（进阶功能）
    参数：
        text: 原文内容
        title: 标题文字
        level: 标题级别（1-6，对应 # 到 ######）
    """
    if level < 1:
        level = 1
    if level > 6:
        level = 6

    hashes = '#' * level
    return f"{hashes} {title}\n\n{text}"


def read_file(file_path):
    """
    读取文件（支持 txt 和 docx）
    新增：支持 Word 文档读取
    """
    if not file_path:
        raise ValueError("文件路径不能为空")

    # 检查文件扩展名
    ext = file_path.lower()

    try:
        # 处理 txt 文件
        if ext.endswith('.txt'):
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()
            return content

        # 处理 docx 文件
        elif ext.endswith('.docx'):
            doc = Document(file_path)
            content = []

            # 读取普通段落
            for paragraph in doc.paragraphs:
                if paragraph.text.strip():
                    content.append(paragraph.text)

            # 读取表格中的文字
            for table in doc.tables:
                for row in table.rows:
                    for cell in row.cells:
                        for paragraph in cell.paragraphs:
                            if paragraph.text.strip():
                                content.append(paragraph.text)

            return '\n'.join(content)

        else:
            raise ValueError(f"不支持的文件格式: {file_path}")

    except FileNotFoundError:
        print(f"错误：文件 {file_path} 不存在")
        return None
    except Exception as e:
        print(f"读取文件失败：{str(e)}")
        return None


def read_and_format(file_path, title=None, title_level=1):
    """
    读取文件并格式化，支持添加标题和字数统计
    增强：支持更多格式，更好的错误处理
    """
    if file_path is None:
        print("错误：文件路径不能为空")
        return None

    # 1. 读取文件
    content = read_file(file_path)
    if content is None:
        return None

    # 2. 清理文本
    cleaned = clean_text(content)

    # 3. 添加标题（如果指定）
    if title:
        cleaned = add_title(cleaned, title, title_level)

    # 4. 字数统计
    word_count = count_words(cleaned)

    # 5. 输出结果
    print("=" * 40)
    print("格式化后的内容：")
    print("=" * 40)
    print(cleaned)
    print("=" * 40)
    print(f"字数统计：{word_count}")
    print("=" * 40)

    return cleaned


# 保存文件功能（新增）
def save_formatted(content, output_path, format_type='txt'):
    """
    保存格式化后的文件
    新增：支持保存为 txt 或 docx
    """
    try:
        if format_type == 'txt':
            if not output_path.endswith('.txt'):
                output_path += '.txt'
            with open(output_path, 'w', encoding='utf-8') as f:
                f.write(content)
            print(f"已保存为纯文本：{output_path}")

        elif format_type == 'docx':
            if not output_path.endswith('.docx'):
                output_path += '.docx'
            doc = Document()
            for line in content.split('\n'):
                if line.strip():
                    doc.add_paragraph(line)
            doc.save(output_path)
            print(f"已保存为Word文档：{output_path}")

        return True
    except Exception as e:
        print(f"保存失败：{str(e)}")
        return False


if __name__ == "__main__":
    # 示例：读取并格式化 test.txt
    result = read_and_format("test.txt", title="我的格式化文本", title_level=1)

    # 如果需要保存
    if result:
        save_formatted(result, "test_formatted")
